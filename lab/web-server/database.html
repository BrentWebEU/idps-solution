Sample Queries for Pentesters

# Connect to database:

mysql -h 172.22.0.3 -u admin -padmin123 company_db


# Get all employees with high salaries:

SELECT first_name, last_name, email, ssn, salary FROM employees WHERE salary > 80000;


# Get all customer credit cards:

SELECT full_name, credit_card, cvv, expiry FROM customers WHERE account_status='active';


# Get all system credentials:

SELECT * FROM system_credentials;


# Get company secrets:

SELECT * FROM company_secrets WHERE classification IN ('TOP_SECRET', 'CONFIDENTIAL');
